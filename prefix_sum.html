Prefix Sum:

The prefix sum of an array is a new array where each element at index i is the sum of the elements from the start of the array to the i-th element in the original array.

Two-Sum Problem (Sorted Array):

                #include <vector>
                #include <iostream>
                using namespace std;
                vector<int> computePrefixSums(const vector<int>& nums) {
                    vector<int> prefixSums(nums.size());
                    prefixSums[0] = nums[0];
                    for (size_t i = 1; i < nums.size(); ++i) {
                        prefixSums[i] = prefixSums[i - 1] + nums[i];
                    }
                    return prefixSums;
                }
                
                int main() {
                    std::vector<int> nums = {1, 2, 3, 4, 5};
                    std::vector<int> prefixSums = computePrefixSums(nums);
                    for (int sum : prefixSums) {
                        std::cout << sum << " ";
                    }
                    return 0;
                }
                    
                    
            
          
            

                public class PrefixSum {
                    public static int[] computePrefixSums(int[] nums) {
                        int[] prefixSums = new int[nums.length];
                        prefixSums[0] = nums[0];
                        for (int i = 1; i < nums.length; i++) {
                            prefixSums[i] = prefixSums[i - 1] + nums[i];
                        }
                        return prefixSums;
                    }
                
                    public static void main(String[] args) {
                        int[] nums = {1, 2, 3, 4, 5};
                        int[] prefixSums = computePrefixSums(nums);
                        for (int sum : prefixSums) {
                            System.out.print(sum + " ");
                        }
                    }
                }

                    prefix_sums = [0] * len(nums)

                    for i in range(1, len(nums)):
                        prefix_sums[i] = prefix_sums[i - 1] + nums[i]
                    return prefix_sums
