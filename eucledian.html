Euclidean algorithms:

The Euclidean algorithm is a way to find the greatest common divisor of two positive integers. GCD of two numbers is the largest number that divides both of them. A simple way to find GCD is to factorize both numbers and multiply common prime factors.

                #include <iostream>
                using namespace std;
                
                // Recursive approach
                int gcdRecursive(int a, int b) {
                    if (b == 0)
                        return a;
                    return gcdRecursive(b, a % b);
                }
                
                // Iterative approach
                int gcdIterative(int a, int b) {
                    while (b != 0) {
                        int temp = b;
                        b = a % b;
                        a = temp;
                    }
                    return a;
                }
                
                int main() {
                    int a = 56, b = 98;
                    cout << "GCD of " << a << " and " << b << " using recursion is " << gcdRecursive(a, b) << endl;
                    cout << "GCD of " << a << " and " << b << " using iteration is " << gcdIterative(a, b) << endl;
                    return 0;
                }

                # Recursive approach
                def gcd_recursive(a, b):
                    if b == 0:
                        return a
                    return gcd_recursive(b, a % b)
                
                # Iterative approach
                def gcd_iterative(a, b):
                    while b != 0:
                        a, b = b, a % b
                    return a
                
                a = 56
                b = 98
                print(f"GCD of {a} and {b} using recursion is {gcd_recursive(a, b)}")
                print(f"GCD of {a} and {b} using iteration is {gcd_iterative(a, b)}")
