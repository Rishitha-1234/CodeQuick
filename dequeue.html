Implementing Deque (Double-Ended Queue) using Arrays :

A deque, short for double-ended queue, is a linear data structure that allows elements to be added or removed from both ends. This versatility makes it suitable for implementing algorithms that require insertion and deletion operations from both front and rear ends efficiently.

                  #include < bits/stdc++.h >

                  class Deque {
                  private:
                      int capacity;
                      int* deque;
                      int front;
                      int rear;
                  
                  public:
                      Deque(int capacity) {
                          this->capacity = capacity;
                          this->deque = new int[capacity];
                          this->front = -1;
                          this->rear = 0;
                      }
                  
                      bool is_empty() {
                          return front == -1;
                      }
                  
                      bool is_full() {
                          return (front == 0 && rear == capacity - 1) || (front == rear + 1);
                      }
                  
                      void insert_front(int item) {
                          if (is_full()) {
                              std::cout << "Deque overflow! Cannot insert at the front." << std::endl;
                          } else {
                              if (front == -1) {
                                  front = rear = 0;
                              } else if (front == 0) {
                                  front = capacity - 1;
                              } else {
                                  front--;
                              }
                              deque[front] = item;
                              std::cout << "Inserted " << item << " at the front of the deque." << std::endl;
                          }
                      }
                  
                      void insert_rear(int item) {
                          if (is_full()) {
                              std::cout << "Deque overflow! Cannot insert at the rear." << std::endl;
                          } else {
                              if (front == -1) {
                                  front = rear = 0;
                              } else if (rear == capacity - 1) {
                                  rear = 0;
                              } else {
                                  rear++;
                              }
                              deque[rear] = item;
                              std::cout << "Inserted " << item << " at the rear of the deque." << std::endl;
                          }
                      }
                  
                      int delete_front() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the front." << std::endl;
                              return -1;
                          } else {
                              int item = deque[front];
                              if (front == rear) {
                                  front = rear = -1;
                              } else if (front == capacity - 1) {
                                  front = 0;
                              } else {
                                  front++;
                              }
                              std::cout << "Deleted " << item << " from the front of the deque." << std::endl;
                              return item;
                          }
                      }
                  
                      int delete_rear() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the rear." << std::endl;
                              return -1;
                          } else {
                              int item = deque[rear];
                              if (front == rear) {
                                  front = rear = -1;
                              } else if (rear == 0) {
                                  rear = capacity - 1;
                              } else {
                                  rear--;
                              }
                              std::cout << "Deleted " << item << " from the rear of the deque." << std::endl;
                              return item;
                          }
                      }
                  
                      void display() {
                          if (is_empty()) {
                              std::cout << "Deque is empty!" << std::endl;
                          } else {
                              std::cout << "Deque elements are:" << std::endl;
                              if (front <= rear) {
                                  for (int i = front; i <= rear; i++) {
                                      std::cout << deque[i] << std::endl;
                                  }
                              } else {
                                  for (int i = front; i < capacity; i++) {
                                      std::cout << deque[i] << std::endl;
                                  }
                                  for (int i = 0; i <= rear; i++) {
                                      std::cout << deque[i] << std::endl;
                                  }
                              }
                          }
                      }
                  
                      ~Deque() {
                          delete[] deque;
                      }
                  };

                class Deque:
                    def __init__(self, capacity):
                        self.capacity = capacity
                        self.deque = [None] * self.capacity
                        self.front = -1
                        self.rear = 0

                    def is_empty(self):
                        return self.front == -1

                    def is_full(self):
                        return (self.front == 0 and self.rear == self.capacity - 1) or (self.front == self.rear + 1)

                    def insert_front(self, item):
                        if self.is_full():
                            print("Deque overflow! Cannot insert at the front.")
                        else:
                            if self.front == -1:
                                self.front = self.rear = 0
                            elif self.front == 0:
                                self.front = self.capacity - 1
                            else:
                                self.front -= 1
                            self.deque[self.front] = item
                            print(f"Inserted {item} at the front of the deque.")

                    def insert_rear(self, item):
                        if self.is_full():
                            print("Deque overflow! Cannot insert at the rear.")
                        else:
                            if self.front == -1:
                                self.front = self.rear = 0
                            elif self.rear == self.capacity - 1:
                                self.rear = 0
                            else:
                                self.rear += 1
                            self.deque[self.rear] = item
                            print(f"Inserted {item} at the rear of the deque.")

                    def delete_front(self):
                        if self.is_empty():
                            print("Deque underflow! Cannot delete from the front.")
                            return None
                        else:
                            item = self.deque[self.front]
                            if self.front == self.rear:
                                self.front = self.rear = -1
                            elif self.front == self.capacity - 1:
                                self.front = 0
                            else:
                                self.front += 1
                            print(f"Deleted {item} from the front of the deque.")
                            return item

                    def delete_rear(self):
                        if self.is_empty():
                            print("Deque underflow! Cannot delete from the rear.")
                            return None
                        else:
                            item = self.deque[self.rear]
                            if self.front == self.rear:
                                self.front = self.rear = -1
                            elif self.rear == 0:
                                self.rear = self.capacity - 1
                            else:
                                self.rear -= 1
                            print(f"Deleted {item} from the rear of the deque.")
                            return item

                    def display(self):
                        if self.is_empty():
                            print("Deque is empty!")
                        else:
                            print("Deque elements are:")
                            if self.front <= self.rear:
                                for i in range(self.front, self.rear + 1):
                                    print(self.deque[i])
                            else:
                                for i in range(self.front, self.capacity):
                                    print(self.deque[i])
                                for i in range(0, self.rear + 1):
                                    print(self.deque[i])

                # Example usage:
                deque = Deque(5)

                deque.insert_front(2)
                deque.insert_rear(3)
                deque.display()
                deque.delete_front()

                deque.display()

                  #include <iostream>

                  class Node {
                  public:
                      int data;
                      Node* next;
                      Node* prev;
                  
                      Node(int data) {
                          this->data = data;
                          this->next = nullptr;
                          this->prev = nullptr;
                      }
                  };
                  
                  class Deque {
                  private:
                      Node* front;
                      Node* rear;
                  
                  public:
                      Deque() {
                          front = nullptr;
                          rear = nullptr;
                      }
                  
                      bool is_empty() {
                          return front == nullptr;
                      }
                  
                      void insert_front(int item) {
                          Node* new_node = new Node(item);
                          if (is_empty()) {
                              front = rear = new_node;
                          } else {
                              new_node->next = front;
                              front->prev = new_node;
                              front = new_node;
                          }
                          std::cout << "Inserted " << item << " at the front of the deque." << std::endl;
                      }
                  
                      void insert_rear(int item) {
                          Node* new_node = new Node(item);
                          if (is_empty()) {
                              front = rear = new_node;
                          } else {
                              rear->next = new_node;
                              new_node->prev = rear;
                              rear = new_node;
                          }
                          std::cout << "Inserted " << item << " at the rear of the deque." << std::endl;
                      }
                  
                      int delete_front() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the front." << std::endl;
                              return -1;
                          } else {
                              int item = front->data;
                              if (front == rear) {
                                  delete front;
                                  front = rear = nullptr;
                              } else {
                                  Node* temp = front;
                                  front = front->next;
                                  front->prev = nullptr;
                                  delete temp;
                              }
                              std::cout << "Deleted " << item << " from the front of the deque." << std::endl;
                              return item;
                          }
                      }
                  
                      int delete_rear() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the rear." << std::endl;
                              return -1;
                          } else {
                              int item = rear->data;
                              if (front == rear) {
                                  delete rear;
                                  front = rear = nullptr;
                              } else {
                                  Node* temp = rear;
                                  rear = rear->prev;
                                  rear->next = nullptr;
                                  delete temp;
                              }
                              std::cout << "Deleted " << item << " from the rear of the deque." << std::endl;
                              return item;

                      }

                          if (is_empty()) {
                              std::cout << "Deque is empty!" << std::endl;
                          } else {
                              std::cout << "Deque elements are:" << std::endl;
                              Node* current = front;
                              while (current != nullptr) {
                                  std::cout << current->data << std::endl;
                                  current = current->next;
                              }
                          }
                      }
                  
                      ~Deque() {
                          while (front != nullptr) {
                              Node* temp = front;
                              front = front->next;
                              delete temp;
                          }
                      }
                  };

                      deque.insert_rear(3);

                      deque.display();
