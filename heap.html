Implementing Min-Heap:

A Heap is a complete binary tree data structure that satisfies the heap property: for every node, the value of its children is less than or equal to its own value. Heaps are usually used to implement priority queues, where the smallest (or largest) element is always at the root of the tree.Building a heap in this manner is more efficient than repeatedly inserting elements into an empty heap. The time complexity for building a heap from an array is O(n), where n is the number of elements in the array.

                  public:
                      MinHeap(const vector<int>& elements) {
                          heap = elements;
                          buildHeap();
                      }
                  
                      void printHeap() const {
                          for (int value : heap) {
                              cout << value << " ";
                          }
                          cout << endl;
                      }
                  
                  private:
                      vector<int> heap;
                  
                      int parent(int i) const {
                          return (i - 1) / 2;
                      }
                  
                      int leftChild(int i) const {
                          return 2 * i + 1;
                      }
                  
                      int rightChild(int i) const {
                          return 2 * i + 2;
                      }
                  
                      void heapifyDown(int i) {
                          int size = heap.size();
                          int smallest = i;
                          int l = leftChild(i);
                          int r = rightChild(i);
                  
                          if (l < size && heap[l] < heap[smallest]) {
                              smallest = l;
                          }
                          if (r < size && heap[r] < heap[smallest]) {
                              smallest = r;
                          }
                          if (smallest != i) {
                              swap(heap[i], heap[smallest]);
                              heapifyDown(smallest);
                          }
                      }
                  
                      void buildHeap() {
                          int size = heap.size();
                          for (int i = size / 2 - 1; i >= 0; --i) {
                              heapifyDown(i);
                          }
                      }
                  };

                import java.util.Arrays;

                public class MinHeap {
                    private int[] heap;
                    private int size;

                    public MinHeap(int[] elements) {
                        heap = Arrays.copyOf(elements, elements.length);
                        size = heap.length;
                        buildHeap();
                    }

                    public void printHeap() {
                        for (int value : heap) {
                            System.out.print(value + " ");
                        }
                        System.out.println();
                    }

                    private int parent(int i) {
                        return (i - 1) / 2;
                    }

                    private int leftChild(int i) {
                        return 2 * i + 1;
                    }

                    private int rightChild(int i) {
                        return 2 * i + 2;
                    }

                    private void heapifyDown(int i) {
                        int smallest = i;
                        int l = leftChild(i);
                        int r = rightChild(i);

                        if (l < size && heap[l] < heap[smallest]) {
                            smallest = l;
                        }
                        if (r < size && heap[r] < heap[smallest]) {
                            smallest = r;
                        }
                        if (smallest != i) {
                            swap(i, smallest);
                            heapifyDown(smallest);
                        }
                    }

                    private void buildHeap() {
                        for (int i = size / 2 - 1; i >= 0; --i) {
                            heapifyDown(i);
                        }
                    }

                    private void swap(int i, int j) {
                        int temp = heap[i];
                        heap[i] = heap[j];
                        heap[j] = temp;
                    }

                    public static void main(String[] args) {
                        int[] elements = {4, 10, 3, 5, 1};
                        MinHeap minHeap = new MinHeap(elements);
                        System.out.print("Min-Heap: ");
                        minHeap.printHeap();
                    }
                }

              
              
            
              
                
                class MinHeap:
                    def __init__(self, elements):
                        self.heap = elements
                        self.build_heap()

                    def parent(self, i):
                        return (i - 1) // 2

                    def left_child(self, i):
                        return 2 * i + 1

                    def right_child(self, i):
                        return 2 * i + 2

                    def heapify_down(self, i):
                        size = len(self.heap)
                        smallest = i
                        l = self.left_child(i)
                        r = self.right_child(i)

                        if l < size and self.heap[l] < self.heap[smallest]:
                            smallest = l
                        if r < size and self.heap[r] < self.heap[smallest]:
                            smallest = r
                        if smallest != i:
                            self.heap[i], self.heap[smallest] = self.heap[smallest], self.heap[i]
                            self.heapify_down(smallest)

                    def build_heap(self):
                        # Start from the last internal node and heapify each node
                        for i in range(len(self.heap) // 2 - 1, -1, -1):
                            self.heapify_down(i)

                    def print_heap(self):
                        print(self.heap)

                # Example usage
                if __name__ == "__main__":
                    elements = [4, 10, 3, 5, 1]
                    min_heap = MinHeap(elements)
                    print("Min-Heap:", end=" ")
                    min_heap.print_heap()
