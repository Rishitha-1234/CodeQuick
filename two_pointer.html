Two Pointer Approach:

The two-pointer technique is a common algorithmic approach used for solving problems related to arrays and linked lists. It involves using two pointers (or indices) to iterate through the data structure simultaneously. This method is especially useful for problems that involve finding pairs or triplets that satisfy a certain condition, merging sorted arrays, or removing duplicates.

                #include <iostream>
                #include <vector>
                
               
                using namespace std;
                
                vector<int> twoSum(vector<int>& nums, int target) {
                    int left = 0, right = nums.size() - 1;
                    while (left < right) {
                        int sum = nums[left] + nums[right];
                        if (sum == target) {
                            return {left, right};
                        } else if (sum < target) {
                            left++;
                        } else {
                            right--;
                        }
                    }
                    return {}; // return empty vector if no pair is found
                }
                
                // Example usage:
                int main() {
                    vector<int> nums = {2, 7, 11, 15};
                    int target = 9;
                    vector<int> result = twoSum(nums, target);
                    if (!result.empty()) {
                        cout << "Indices: " << result[0] << ", " << result[1] << endl;
                    } else {
                        cout << "No pair found." << endl;
                    }
                    return 0;
                }

            def two_sum(nums, target):
                left, right = 0, len(nums) - 1
                while left < right:
                    total = nums[left] + nums[right]
                    if total == target:
                        return [left, right]
                    elif total < target:
                        left += 1
                    else:
                        right -= 1
                return []

Removing Duplicates from a Sorted Array:

                #include <iostream>
                #include <vector>
                #include <stack>
                #include <unordered_set>
                using namespace std;
                
                int removeDuplicates(vector<int>& nums) {
                    if (nums.empty()) return 0;
                    int index = 1;
                    for (int i = 1; i < nums.size(); i++) {
                        if (nums[i] != nums[i - 1]) {
                            nums[index++] = nums[i];
                        }
                    }
                    return index;
                }
                
                // Example usage:
                int main() {
                    vector<int> nums = {1, 1, 2, 2, 3, 3, 4};
                    int newLength = removeDuplicates(nums);
                    cout << "New length: " << newLength << endl;
                    for (int i = 0; i < newLength; i++) {
                        cout << nums[i] << " ";
                    }
                    cout << endl;
                    return 0;
                }

            def remove_duplicates(nums):
                if not nums:
                    return 0
                index = 1
                for i in range(1, len(nums)):
                    if nums[i] != nums[i - 1]:
                        nums[index] = nums[i]
                        index += 1
                return index
            
            # Example usage:
            nums = [1, 1, 2, 2, 3, 3, 4]
            new_length = remove_duplicates(nums)
            print(f"New length: {new_length}")
            print(nums[:new_length])
