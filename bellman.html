Bellman-Ford Algorithm:

The Bellman-Ford algorithm is designed to find the shortest paths from a single source vertex to all other vertices in a weighted graph, even when some of the edge weights are negative. It accomplishes this by iteratively relaxing all edges in the graph over Vâˆ’1 passes, where V is the number of vertices.

                struct Edge {
                    int src, dest, weight;

                class Solution {
                public:
                    // Function to find the shortest distances from the source vertex
                    // to all other vertices using Bellman-Ford algorithm
                    vector<int> bellmanFord(int V, vector<Edge>& edges, int src) {
                        vector<int> dist(V, INT_MAX); // Initialize distances with infinity
                        dist[src] = 0; // Distance from source to itself is zero
                
                        // Relax all edges (V-1) times to ensure shortest path is found
                        for (int i = 1; i < V; ++i) {
                            for (const auto& edge : edges) {
                                int u = edge.src;
                                int v = edge.dest;
                                int weight = edge.weight;
                                if (dist[u] != INT_MAX && dist[u] + weight < dist[v]) {
                                    dist[v] = dist[u] + weight;
                                }
                            }
                        }
                
                        // Check for negative-weight cycles
                        for (const auto& edge : edges) {
                            int u = edge.src;
                            int v = edge.dest;
                            int weight = edge.weight;
                            if (dist[u] != INT_MAX && dist[u] + weight < dist[v]) {
                                // Negative weight cycle found, return empty vector or handle as needed
                                return {};
                            }
                        }
                
                        return dist;
                    }
                };

                from typing import List

                class Edge:
                    def __init__(self, src: int, dest: int, weight: int):
                        self.src = src
                        self.dest = dest
                        self.weight = weight

                class Solution:
                    def bellmanFord(self, V: int, edges: List[Edge], src: int) -> List[int]:
                        dist = [float('inf')] * V
                        dist[src] = 0

                        for _ in range(V - 1):
                            for edge in edges:
                                if dist[edge.src] != float('inf') and dist[edge.src] + edge.weight < dist[edge.dest]:
                                    dist[edge.dest] = dist[edge.src] + edge.weight

                        # Check for negative-weight cycles
                        for edge in edges:
                            if dist[edge.src] != float('inf') and dist[edge.src] + edge.weight < dist[edge.dest]:
                                print("Graph contains negative weight cycle")
                                return []

                        return dist

                if __name__ == "__main__":

Removing Duplicates from a Sorted Array:

                #include <iostream>
                #include <vector>

                int removeDuplicates(vector<int>& nums) {
                    if (nums.empty()) return 0;
                    int index = 1;
                    for (int i = 1; i < nums.size(); i++) {
                        if (nums[i] != nums[i - 1]) {
                            nums[index++] = nums[i];
                        }
                    }
                    return index;
                }
                
                // Example usage:
                int main() {
                    vector<int> nums = {1, 1, 2, 2, 3, 3, 4};
                    int newLength = removeDuplicates(nums);
                    cout << "New length: " << newLength << endl;
                    for (int i = 0; i < newLength; i++) {
                        cout << nums[i] << " ";
                    }
                    cout << endl;
                    return 0;
                }

            def remove_duplicates(nums):
                if not nums:
                    return 0
                index = 1
                for i in range(1, len(nums)):
                    if nums[i] != nums[i - 1]:
                        nums[index] = nums[i]
                        index += 1
                return index
