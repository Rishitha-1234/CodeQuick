Hare and Tortoise:

The "hare and tortoise" algorithm, also known as Floyd's cycle detection algorithm, is a popular algorithm used to detect cycles in a linked list or to find the starting point of a cycle.The tortoise (slow pointer) eventually catches up to the hare (fast pointer). The hare and tortoise algorithm is efficient with a Time complexity of O(n) and a constant space complexity O(1), making it suitable for large datasets or when memory usage needs to be minimal.

                // Definition for a singly linked list node
                struct ListNode {
                    int val;
                    ListNode *next;
                    ListNode(int x) : val(x), next(nullptr) {}
                };
                
                ListNode* hareTortoise(ListNode* head) {
                    ListNode *tortoise = head;
                    ListNode *hare = head;
                    
                    // Phase 1: detect cycle
                    while (hare != nullptr && hare->next != nullptr) {
                        tortoise = tortoise->next;
                        hare = hare->next->next;
                        
                        if (tortoise == hare)
                            break;
                    }
                    
                    // No cycle detected
                    if (hare == nullptr || hare->next == nullptr)
                        return nullptr;
                    
                    // Phase 2: find the start of the cycle
                    tortoise = head;
                    while (tortoise != hare) {
                        tortoise = tortoise->next;
                        hare = hare->next;
                    }
                    
                    return hare;
                }

                    cout << "nullptr" << std::endl;
                }

                # Definition for a singly linked list node
                class ListNode:
                    def __init__(self, x):
                        self.val = x
                        self.next = None
                
                def hare_tortoise(head):
                    tortoise = head
                    hare = head
                    
                    # Phase 1: detect cycle
                    while hare and hare.next:
                        tortoise = tortoise.next
                        hare = hare.next.next
                        
                        if tortoise == hare:
                            break
                    
                    # No cycle detected
                    if not hare or not hare.next:
                        return None
                    
                    # Phase 2: find the start of the cycle
                    tortoise = head
                    while tortoise != hare:
                        tortoise = tortoise.next
                        hare = hare.next
                    
                    return hare
                
                # function to print the linked list

                    current = head

                        current = current.next
