Implementing Deque (Double-Ended Queue) using Arrays :

A hash table, also known as a hash map, is a data structure that implements an associative array abstract data type, a structure that can map keys to values. It uses a hash function to compute an index into an array of buckets or slots, from which the desired value can be found.

                  #include < iostream >
                  #include < vector >
                  #include < list >
                  using namespace std;
                    class HashTable {
                    private:
                        int capacity;
                        vector<list<pair<string, int>>> table;
                    
                    public:
                        HashTable(int capacity) : capacity(capacity), table(capacity) {}
                    
                        int hash_func(const std::string& key) {
                            // Simple hash function using ASCII values
                            int hash = 0;
                            for (char ch : key) {
                                hash += int(ch);
                            }
                            return hash % capacity;
                        }
                    
                        void insert(string& key, int value) {
                            int index = hash_func(key);
                            for (auto& entry : table[index]) {
                                if (entry.first == key) {
                                    entry.second = value;
                                    return;
                                }
                            }
                            table[index].emplace_back(key, value);
                        }
                    
                        void remove(string& key) {
                            int index = hash_func(key);
                            auto& bucket = table[index];
                            for (auto it = bucket.begin(); it != bucket.end(); ++it) {
                                if (it->first == key) {
                                    bucket.erase(it);
                                    return;
                                }
                            }
                        }
                    
                        int search(string& key) {
                            int index = hash_func(key);
                            for (auto& entry : table[index]) {
                                if (entry.first == key) {
                                    return entry.second;
                                }
                            }
                            return -1; // Return -1 or throw exception for not found
                        }
                    
                        void display() {
                            for (int i = 0; i < capacity; ++i) {
                                cout << "[" << i << "]: ";
                                for (auto& entry : table[i]) {
                                    cout << "(" << entry.first << ", " << entry.second << ") ";
                                }
                                cout << endl;
                            }
                        }
                    };
                    
                    int main() {
                        HashTable hash_table(10);
                        hash_table.insert("apple", 5);
                        hash_table.insert("banana", 2);
                        hash_table.insert("orange", 7);
                    
                        std::cout << "Value of 'banana': " << hash_table.search("banana") << std::endl;
                        hash_table.remove("banana");
                        std::cout << "Value of 'banana' after removal: " << hash_table.search("banana") << std::endl;
                    
                        hash_table.display();
                    
                        return 0;
                    }
                    
                  
                  
              
            
              
                public class Deque {
                  private int capacity;
                  private int[] deque;
                  private int front;
                  private int rear;
              
                  public Deque(int capacity) {
                      this.capacity = capacity;
                      this.deque = new int[capacity];
                      this.front = -1;
                      this.rear = 0;
                  }
              
                  public boolean isEmpty() {
                      return front == -1;
                  }
              
                  public boolean isFull() {
                      return (front == 0 && rear == capacity - 1) || (front == rear + 1);
                  }
              
                  public void insertFront(int item) {
                      if (isFull()) {
                          System.out.println("Deque overflow! Cannot insert at the front.");
                      } else {
                          if (front == -1) {
                              front = rear = 0;
                          } else if (front == 0) {
                              front = capacity - 1;
                          } else {
                              front--;
                          }
                          deque[front] = item;
                          System.out.println("Inserted " + item + " at the front of the deque.");
                      }
                  }
              
                  public void insertRear(int item) {
                      if (isFull()) {
                          System.out.println("Deque overflow! Cannot insert at the rear.");
                      } else {
                          if (front == -1) {
                              front = rear = 0;
                          } else if (rear == capacity - 1) {
                              rear = 0;
                          } else {
                              rear++;
                          }
                          deque[rear] = item;
                          System.out.println("Inserted " + item + " at the rear of the deque.");
                      }
                  }
              
                  public int deleteFront() {
                      if (isEmpty()) {
                          System.out.println("Deque underflow! Cannot delete from the front.");
                          return -1;
                      } else {
                          int item = deque[front];
                          if (front == rear) {
                              front = rear = -1;
                          } else if (front == capacity - 1) {
                              front = 0;
                          } else {
                              front++;
                          }
                          System.out.println("Deleted " + item + " from the front of the deque.");
                          return item;
                      }
                  }
              
                  public int deleteRear() {
                      if (isEmpty()) {
                          System.out.println("Deque underflow! Cannot delete from the rear.");
                          return -1;
                      } else {
                          int item = deque[rear];
                          if (front == rear) {
                              front = rear = -1;
                          } else if (rear == 0) {
                              rear = capacity - 1;
                          } else {
                              rear--;
                          }
                          System.out.println("Deleted " + item + " from the rear of the deque.");
                          return item;
                      }
                  }
              
                  public void display() {
                      if (isEmpty()) {
                          System.out.println("Deque is empty!");
                      } else {
                          System.out.println("Deque elements are:");
                          if (front <= rear) {
                              for (int i = front; i <= rear; i++) {
                                  System.out.println(deque[i]);
                              }
                          } else {
                              for (int i = front; i < capacity; i++) {
                                  System.out.println(deque[i]);
                              }
                              for (int i = 0; i <= rear; i++) {
                                  System.out.println(deque[i]);
                              }
                          }
                      }
                  }
              
                  public static void main(String[] args) {
                      Deque deque = new Deque(5);
                      deque.insertFront(1);
                      deque.insertFront(2);
                      deque.insertRear(3);
                      deque.display();
                      deque.deleteFront();
                      deque.deleteRear();
                      deque.display();
                  }
              }
                
              
              
            
              
                
                class HashTable:
                    def __init__(self, capacity):
                        self.capacity = capacity
                        self.table = [[] for _ in range(capacity)]

                    def hash_func(self, key):
                        return hash(key) % self.capacity

                    def insert(self, key, value):
                        index = self.hash_func(key)
                        for entry in self.table[index]:
                            if entry[0] == key:
                                entry[1] = value
                                return
                        self.table[index].append([key, value])

                    def delete(self, key):
                        index = self.hash_func(key)
                        bucket = self.table[index]
                        for i in range(len(bucket)):
                            if bucket[i][0] == key:
                                del bucket[i]
                                return

                    def search(self, key):
                        index = self.hash_func(key)
                        bucket = self.table[index]
                        for entry in bucket:
                            if entry[0] == key:
                                return entry[1]
                        return None

                # Example usage:
                hash_table = HashTable(10)
                hash_table.insert("apple", 5)
                hash_table.insert("banana", 2)
                hash_table.insert("orange", 7)

                print(hash_table.search("banana"))  # Output: 2

                hash_table.delete("banana")
                print(hash_table.search("banana"))  # Output: None

                  class Node {
                  public:
                      int data;
                      Node* next;
                      Node* prev;
                  
                      Node(int data) {
                          this->data = data;
                          this->next = nullptr;
                          this->prev = nullptr;
                      }
                  };
                  
                  class Deque {
                  private:
                      Node* front;
                      Node* rear;
                  
                  public:
                      Deque() {
                          front = nullptr;
                          rear = nullptr;
                      }
                  
                      bool is_empty() {
                          return front == nullptr;
                      }
                  
                      void insert_front(int item) {
                          Node* new_node = new Node(item);
                          if (is_empty()) {
                              front = rear = new_node;
                          } else {
                              new_node->next = front;
                              front->prev = new_node;
                              front = new_node;
                          }
                          std::cout << "Inserted " << item << " at the front of the deque." << std::endl;
                      }
                  
                      void insert_rear(int item) {
                          Node* new_node = new Node(item);
                          if (is_empty()) {
                              front = rear = new_node;
                          } else {
                              rear->next = new_node;
                              new_node->prev = rear;
                              rear = new_node;
                          }
                          std::cout << "Inserted " << item << " at the rear of the deque." << std::endl;
                      }
                  
                      int delete_front() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the front." << std::endl;
                              return -1;
                          } else {
                              int item = front->data;
                              if (front == rear) {
                                  delete front;
                                  front = rear = nullptr;
                              } else {
                                  Node* temp = front;
                                  front = front->next;
                                  front->prev = nullptr;
                                  delete temp;
                              }
                              std::cout << "Deleted " << item << " from the front of the deque." << std::endl;
                              return item;
                          }
                      }
                  
                      int delete_rear() {
                          if (is_empty()) {
                              std::cout << "Deque underflow! Cannot delete from the rear." << std::endl;
                              return -1;
                          } else {
                              int item = rear->data;
                              if (front == rear) {
                                  delete rear;
                                  front = rear = nullptr;
                              } else {
                                  Node* temp = rear;
                                  rear = rear->prev;
                                  rear->next = nullptr;
                                  delete temp;

                              Node* current = front;
                              while (current != nullptr) {
                                  std::cout << current->data << std::endl;
                                  current = current->next;
                              }

                          while (front != nullptr) {
                              Node* temp = front;
                              front = front->next;
                              delete temp;
                          }
                      }
